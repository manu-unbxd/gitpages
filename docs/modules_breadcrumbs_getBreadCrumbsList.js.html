/**
*  @description 
* will return breadcrumbs array if it is available. in following format.
   <pre>
        <code>
        [
            {
                "level": 1,
                "filterField": "categoryPath",
                "value": "Dress"
            },
            {
                "level": 2,
                "filterField": "categoryPath",
                "value": "Short Sleeves"
            }
            ]
        </code>
    </pre>
    * @method getBreadCrumbsList 
*/
export default function(){
    const data = this.getBreadCrumbs();
    let arr = [];
    let temp = data;
    for(let i=0;i<this.options.facetDepth;i++){
        if(temp.level) {
            const {
                level,
                values,
                filterField
            } = temp;
            arr.push({
                level,
                filterField,
                value:values[0].value
            })
        }
        if(temp.child) {
            temp = temp.child;  
        } else {
            break;
        }
    }
    return arr;
};
